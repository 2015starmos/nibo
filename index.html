// ========== 全域變數 ========== let petList = JSON.parse(localStorage.getItem('petList')) || []; let isPetEdit = false; // ========== 頁面切換 ========== function showPage(page) { document.querySelectorAll('.page-content').forEach(el => el.style.display = 'none'); document.getElementById('page-' + page).style.display = 'block'; if (page === 'pet-manage') renderPetTable(); } // ========== 寵物管理：彈窗 ========== function openPetModal(id = null) { isPetEdit = !!id; document.getElementById('pet-modal-title').textContent = isPetEdit ? '修改寵物資料' : '新增寵物資料'; document.getElementById('edit-pet-id').value = id || ''; if (isPetEdit) { const pet = petList.find(p => p.id === id); document.getElementById('pet-name').value = pet.name; document.getElementById('owner-name').value = pet.ownerName; document.getElementById('owner-phone').value = pet.ownerPhone; document.getElementById('pet-chip').value = pet.chip || ''; } else { document.getElementById('pet-name').value = ''; document.getElementById('owner-name').value = ''; document.getElementById('owner-phone').value = ''; document.getElementById('pet-chip').value = ''; } document.getElementById('pet-modal').style.display = 'flex'; } function closePetModal() { document.getElementById('pet-modal').style.display = 'none'; } // ========== 寵物：儲存 ========== function savePet() { const name = document.getElementById('pet-name').value.trim(); const ownerName = document.getElementById('owner-name').value.trim(); const ownerPhone = document.getElementById('owner-phone').value.trim(); const chip = document.getElementById('pet-chip').value.trim(); const id = document.getElementById('edit-pet-id').value; if (!name || !ownerName || !ownerPhone) { alert('請填寫必填欄位！'); return; } if (isPetEdit) { const idx = petList.findIndex(p => p.id == id); petList[idx] = { ...petList[idx], name, ownerName, ownerPhone, chip }; } else { petList.push({ id: Date.now(), name, ownerName, ownerPhone, chip }); } localStorage.setItem('petList', JSON.stringify(petList)); renderPetTable(); closePetModal(); alert(isPetEdit ? '已更新！' : '已新增！'); } // ========== 寵物：渲染表格 ========== function renderPetTable(list = petList) { const tbody = document.getElementById('pet-table-body'); tbody.innerHTML = ''; list.forEach(pet => { tbody.innerHTML += ` ${pet.name} ${pet.ownerName} ${pet.ownerPhone} ${pet.chip || '-'} `; }); } // ========== 寵物：搜尋 ========== function searchPet() { const kw = document.getElementById('pet-search').value.trim().toLowerCase(); const filtered = petList.filter(p => p.name.toLowerCase().includes(kw) || p.ownerPhone.includes(kw) ); renderPetTable(filtered); } // ========== 寵物：刪除 ========== function deletePet(id) { if (!confirm('確定要刪除嗎？')) return; petList = petList.filter(p => p.id != id); localStorage.setItem('petList', JSON.stringify(petList)); renderPetTable(); } // ========== 預設渲染 ========== renderPetTable();
